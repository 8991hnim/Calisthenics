CREATE TABLE category
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Name TEXT,
    Background TEXT
)

CREATE TABLE post
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Title TEXT,
    Content TEXT,
    ShortContent TEXT,
    LinkYoutube TEXT,
    Image TEXT,
    Background TEXT,
    CategoryID INT,
    FOREIGN KEY (CategoryID) REFERENCES category(ID)
)

CREATE TABLE `user`
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Account TEXT,
    Pass TEXT,
    Username TEXT,
    Type INT
)

CREATE TABLE exercise
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Name TEXT,
    Gif TEXT
)


CREATE TABLE training
(
 	LevelID INT,
    Day INT,
    ExerciseID INT,
    `Set` INT,
    `Rep` INT,
    BreakTime TEXT,
    PRIMARY KEY (LevelID, Day, ExerciseID), 
    FOREIGN KEY (ExerciseID) REFERENCES exercise(ID)
)



CREATE TABLE progress_training
(
    UserID INT,
    LevelID INT,
    DayTrained INT,
    FOREIGN KEY (UserID) REFERENCES `user`(ID), 
    FOREIGN KEY (LevelID) REFERENCES training(LevelID),
    PRIMARY KEY (UserID, LevelID)
)
