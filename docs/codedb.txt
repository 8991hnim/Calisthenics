CREATE TABLE category
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Name TEXT
)

CREATE TABLE post
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Title TEXT,
    Content TEXT,
    ShortContent TEXT,
    LinkYoutube TEXT,
    Image TEXT,
    CategoryID INT,
    FOREIGN KEY (CategoryID) REFERENCES category(ID)
)

CREATE TABLE `user`
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Account TEXT,
    Pass TEXT,
    Username TEXT,
    Email TEXT,
    Type INT
)

CREATE TABLE exercise
(
 	ID INT PRIMARY KEY AUTO_INCREMENT,
    Name varchar(100),
    UrlYT TEXT
)

CREATE TABLE Level
(
	ID INT PRIMARY KEY AUTO_INCREMENT
)

CREATE TABLE Day
(
	ID INT PRIMARY KEY AUTO_INCREMENT,
    	Day INT
)

CREATE TABLE training
(
    LevelID INT,
    DayID INT,
    ExerciseID INT,
    `Set` INT,
    `Rep` INT,
    BreakTime TEXT,
    FOREIGN KEY (ExerciseID) REFERENCES exercise(ID),
    FOREIGN KEY (DayID) REFERENCES day(ID),
    FOREIGN KEY (LevelID) REFERENCES `level`(ID),
    PRIMARY KEY (LevelID, DayID, ExerciseID)
)



CREATE TABLE progress_training
(
    UserID INT,
    LevelID INT,
    DayTrained INT,
    FOREIGN KEY (UserID) REFERENCES `user`(ID), 
    FOREIGN KEY (LevelID) REFERENCES `level`(ID),
    PRIMARY KEY (UserID, LevelID)
)
